[ check kvm ]
lsmod | grep kvm


[ create disk image ]
qemu-img create -f <image_option> <image_name> <size_in_gb>

< image option >
- raw
- qcow2 (qemu copy on write)

ex) qemu-img create -f qcow2 arch_vm.qcow2 100G


[ start vm ]
qemu-system-x86_64 -enable-kvm -smp <number_of_cpu> -m <mamory_size> -cdrom <iso_image> -boot order=d <disk_image_name>
(-boot order=d means boot cdrom first)

ex) qemu-system-x86_64 -enable-kvm -smp 1 -m 2G -cdrom arch.iso -boot order=d arch_vm.qcow2
ex) qemu-system-x86_64 -m 2G -enable-kvm arch_vm.qcow2
